	a[0] = f;
	a[1] = f_name;
	return a;
}

	return getScriptVar("all_plot_flags");

function getFlagLists(plot) {
	flags = plot.getFlags(plot);
	all_flags = getAllPlotFlags();
	flags_true = list.new();
	flags_false = list.new();
	
	iter = iterator(all_flags);
	while(hasNext(iter)) {
		a = next(iter);
		f = a[0];
		if(flags >= f) {
			list.add(flags_true, a);
			flags -= f;
		} else {
			list.add(flags_false, a);
		}
	}
	a = array.new(2);
	a[0] = flags_true;
	a[1] = flags_false;
	return a;
}
