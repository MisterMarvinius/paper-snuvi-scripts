function createMainMenu(inv_name, market, quests, skills) {
	inv = inv.new("222222222", text.new(inv_name));
	index = 0;
	if(market) {
		inv.setItem(inv, index++, $emerald);
	}
	if(quests) {
		inv.setItem(inv, index++, $quest_book);
	}
	if(skills) {
		inv.setItem(inv, index++, $scroll);
	}
	return inv;
}
